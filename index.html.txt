<script>
const sentences = [
  {
    level: "B1",
    text: "I realized that silence is sometimes the best answer",
    words: {
      realized: "fark ettim",
      silence: "sessizlik",
      sometimes: "bazen",
      answer: "cevap"
    }
  },
  {
    level: "C1",
    text: "People often underestimate how much emotional fatigue affects decisions",
    words: {
      underestimate: "küçümsemek",
      emotional: "duygusal",
      fatigue: "yorgunluk",
      affects: "etkiler",
      decisions: "kararlar"
    }
  }
];

let index = 0;

function render() {
  document.getElementById("level").innerText = "Level: " + sentences[index].level;
  const container = document.getElementById("sentence");
  container.innerHTML = "";

  sentences[index].text.split(" ").forEach(word => {
    const clean = word.toLowerCase();
    const span = document.createElement("span");
    span.className = "word";
    span.innerText = word;
    span.onclick = () => {
      alert(sentences[index].words[clean] || "Bu kelime için anlam yok");
    };
    container.appendChild(span);
  });
}

function nextSentence() {
  index = (index + 1) % sentences.length;
  render();
}

render();
</script>
